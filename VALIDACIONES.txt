.::VALIDAR CEDULA::.
function validarCedula(e) {
    // Obtener la longitud actual de la cédula
    var cedula = document.getElementById("ced").value;
    // Obtener el código ASCII de la tecla presionada
    var charCode = e.keyCode;
    // Permitir las teclas de control como "Enter" y "Backspace"
    if (charCode == 13 || charCode == 8) {
        return true;
    }
    // Convertir el código ASCII a carácter
    var charTyped = String.fromCharCode(charCode);
    // Validar si el carácter ingresado es un número y si la longitud actual es menor que 10
    if (/\d/.test(charTyped) && cedula.length < 10) {
        return true;
    } else {
        // Si no es un número o la longitud es 10, prevenir la acción por defecto
        e.preventDefault();
        return false;
    }
}

VALIDAR EDAD

function validarEdad(e){
	var inputEdad = document.getElementById("edad").value;
	var filtro = inputEdad.replace(/[^0-9]/g, "");
	var edad = parseInt(filtro, 10);
	var mensaje_error = document.getElementById("msm");

	if (edad > 100) {
		e.preventDefault();
		document.getElementById("edad").value = "";
		mensaje_error.innerText = "La edad no debe de ser mayor a 100";
	}else{
		mensaje_error.innerText = "";
		if (filtro !== inputEdad) {
			e.preventDefault();
			document.getElementById("edad").value = filtro;
		}
	}
}
document.getElementById("edad").addEventListener("input", validarEdad);
